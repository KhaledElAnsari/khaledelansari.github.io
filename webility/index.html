Webility.js

The Front-End developers Utility Belt

Webility.js is a lightweight, smooth, easy-to-use an zero-dependencies library to ease the pain of some Front-End development issues.

Below you can see the availabe functionalities with some examples.

Webility APIs

You can check the list of availabe function at the moment at the Github Repository.

Example

Here's a Fast Full example with all of the available functions at the moment, check your browser console to see the results. 
        webility.pageready(function() {
          var str = "john doe";
          var obj = {"khaled": 24, "max": 19, "nadia": 27};
          var arr = [0,7,5,1,9];

          console.log(webility.title(str));
          console.log(webility.capitalize(str));
          console.log(webility.quote(str));

          var today = webility.today();
          console.log(today);
          console.log(webility.toUTC("2016-08-17T10:00"));
          console.log(webility.toLocalTime("2016-08-17T10:00"));
          console.log(webility.dayAfter(today, 4));
          console.log(webility.getDays("2016-08-17", "2016-08-21"));

          console.log(webility.sort(arr));
          console.log(webility.objToArray(obj));
          console.log(webility.isEmpty(obj));
          console.log(webility.get(obj, "Mohammad", 50));

          console.log(webility.browser());
          console.log(webility.os());
          console.log(webility.device());
          console.log(webility.ismobile());
          console.log(webility.uuid());

          var elems = webility.select("#eg");
          console.log(elems[0]);
          console.log(elems.ID());
          console.log(elems.addclass("anotherclass"));
          console.log(elems.hasclass("someclass"));
          console.log(elems.removeclass("anotherclass"));
        });

That's it, the function will be updated from time to time and your contribution is always welcomed.
